696. 计数二进制子串

连续值计数技巧 
class Solution:
    def countBinarySubstrings(self, s: str) -> int:
        res = 0
        cnts = []
        pre = '#'
        for c in s:
            if c == pre: cnts[-1] += 1
            else: cnts.append(1)
            pre = c
        for i in range(1, len(cnts)):
            res += min(cnts[i-1], cnts[i])
        return res
